新增inbody table
CREATE TABLE `startfitness`.`inbody` (`id` INT NOT NULL AUTO_INCREMENT , `name` VARCHAR(100) NOT NULL , `tel` VARCHAR(100) NOT NULL , `email` VARCHAR(256) NOT NULL , `gym` VARCHAR(100) NOT NULL , `date` DATE NOT NULL , `time` TIME NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;

新增membertable
CREATE TABLE `startfitness`.`member` (`id` BIGINT(20) NOT NULL AUTO_INCREMENT , `name` VARCHAR(256) NOT NULL , `email` VARCHAR(256) NOT NULL , `tel` VARCHAR(128) NULL , `account` VARCHAR(128) NOT NULL , `password` VARCHAR(256) NOT NULL , `status` TINYINT(1) NOT NULL DEFAULT '1' , `permission` VARCHAR(50) NOT NULL DEFAULT 'member' , PRIMARY KEY (`id`)) ENGINE = InnoDB;
